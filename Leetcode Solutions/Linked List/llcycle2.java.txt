/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode detectCycle(ListNode head) {
        if(head==null||head.next==null)
        {
            return null;
        }
        ListNode fast = head;
        ListNode slow = head;
        ListNode temp = head;
        while(fast!=null&&fast.next!=null)
        {
            fast = fast.next.next;
            slow = slow.next;
            if(fast==slow)
            {
                while(temp!=fast)
                {
                    temp = temp.next;
                    fast = fast.next;
                }
            return temp;
            }
        }
        return null;
        
    }
}
